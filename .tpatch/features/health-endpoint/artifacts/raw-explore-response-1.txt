## Minimal Changeset

### 1. Create `src/lib/uptime.ts`
```typescript
// Track server startup time for uptime calculation
let startTime: number;

export function initializeUptime(): void {
  startTime = Date.now();
}

export function getUptimeSeconds(): number {
  if (!startTime) {
    return 0;
  }
  return Math.floor((Date.now() - startTime) / 1000);
}
```

### 2. Create `src/lib/version.ts`
```typescript
import { readFileSync } from 'fs';
import { join } from 'path';

let cachedVersion: string;

function loadVersion(): string {
  try {
    const packageJsonPath = join(process.cwd(), 'package.json');
    const packageJson = JSON.parse(readFileSync(packageJsonPath, 'utf-8'));
    return packageJson.version || 'unknown';
  } catch (error) {
    console.warn('Failed to read version from package.json:', error);
    return 'unknown';
  }
}

export function getVersion(): string {
  if (!cachedVersion) {
    cachedVersion = loadVersion();
  }
  return cachedVersion;
}
```

### 7. Create `tests/health.unit.test.ts`
```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest'; // or jest
import { getUptimeSeconds, initializeUptime } from '~/lib/uptime';
import { getVersion } from '~/lib/version';
import { getModelCount } from '~/lib/model-counter';

describe('Health Module', () => {
  describe('uptime', () => {
    it('should return numeric uptime in seconds', () => {
      initializeUptime();
      const uptime = getUptimeSeconds();
      expect(typeof uptime).toBe('number');
      expect(uptime).toBeGreaterThanOrEqual(0);
    });

    it('uptime should increase between calls', async () => {
      initializeUptime();
      const uptime1 = getUptimeSeconds();
      await new Promise(resolve => setTimeout(resolve, 100));
      const uptime2 = getUptimeSeconds();
      expect(uptime2).toBeGreaterThanOrEqual(uptime1);
    });
  });

  describe('version', () => {
    it('should return a non-empty string', () => {
      const version = getVersion();
      expect(typeof version).toBe('string');
      expect(version.length).toBeGreaterThan(0);
    });

    it('should cache version after first call', () => {
      const version1 = getVersion();
      const version2 = getVersion();
      expect(version1).toBe(version2);
    });
  });

  describe('model counter', () => {
    it('should return non-negative integer', () => {
      const count = getModelCount();
      expect(Number.isInteger(count)).toBe(true);
      expect(count).toBeGreaterThanOrEqual(0);
    });
  });
});
```
