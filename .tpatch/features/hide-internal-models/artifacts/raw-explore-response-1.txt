## Minimal Changeset

### `src/lib/filter-models.ts` (new file)
```ts
export function isInternalModel(id: string): boolean {
  return id.includes('internal') || id.startsWith('accounts/');
}

export function filterModels<T extends { id: string }>(models: T[], hideInternal: boolean): T[] {
  if (!hideInternal) return models;
  return models.filter(m => !isInternalModel(m.id));
}
```
