// src/routes/models/route.ts

import { Hono } from 'hono';
import { MODEL_FAMILIES } from '../../lib/model-mapping';
import { config } from '../../lib/state'; // Import config

const models = new Hono();

models.get('/', (c) => {
  let availableModels = MODEL_FAMILIES;

  if (config.modelFilter) {
    const filter = config.modelFilter.toLowerCase();
    availableModels = MODEL_FAMILIES.filter(
      (model) => model.vendor?.toLowerCase() === filter,
    );
  }

  return c.json({
    object: 'list',
    data: availableModels.map((model) => ({
      id: model.id,
      object: 'model',
      created: Math.floor(Date.now() / 1000),
      owned_by: model.vendor || 'unknown',
    })),
  });
});

export default models;
